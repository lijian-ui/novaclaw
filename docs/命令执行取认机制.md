### 3. 新增全局确认管理

创建新文件 `backend/src/tools/approval.rs`：

```rust
use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::RwLock;
use crate::tools::types::{ApprovalRequired, ToolExecutionStatus};

/// 等待确认的操作
struct PendingApproval {
    pub approval: ApprovalRequired,
    pub session_id: String,
    pub tool_name: String,
    pub arguments: String,
    pub created_at: chrono::DateTime<chrono::Utc>,
}

/// 确认管理器
pub struct ApprovalManager {
    pending: RwLock<HashMap<String, PendingApproval>>,
}

impl ApprovalManager {
    pub fn new() -> Self {
        Self {
            pending: RwLock::new(HashMap::new()),
        }
    }

    /// 记录待确认操作
    pub async fn add_pending(
        &self,
        approval_id: String,
        approval: ApprovalRequired,
        session_id: String,
        tool_name: String,
        arguments: String,
    ) {
        let pending = PendingApproval {
            approval,
            session_id,
            tool_name,
            arguments,
            created_at: chrono::Utc::now(),
        };
        self.pending.write().await.insert(approval_id, pending);
    }

    /// 检查是否有待确认
    pub async fn get_pending(&self, approval_id: &str) -> Option<ApprovalRequired> {
        self.pending.read().await.get(approval_id).map(|p| p.approval.clone())
    }

    /// 移除已处理的确认
    pub async fn remove_pending(&self, approval_id: &str) {
        self.pending.write().await.remove(approval_id);
    }

    /// 清理超时的确认（比如 5 分钟）
    pub async fn cleanup_expired(&self) {
        let now = chrono::Utc::now();
        let mut pending = self.pending.write().await;
        pending.retain(|_, p| (now - p.created_at) < chrono::Duration::minutes(5));
    }
}
```
